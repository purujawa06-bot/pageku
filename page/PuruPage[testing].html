// Menggunakan module.exports untuk mengekspor fungsi module.exports = { getTes: (req, res) => { // req.query digunakan untuk mengambil parameter dari URL (setelah tanda ?) // Contoh URL: http://localhost:3000/api?tes=halo const nilaiTes = req.query.tes; if (nilaiTes) { res.status(200).json({ pesan: "Data berhasil diterima", data_tes: nilaiTes }); } else { res.status(400).json({ pesan: "Parameter 'tes' tidak ditemukan di URL" }); } } };
